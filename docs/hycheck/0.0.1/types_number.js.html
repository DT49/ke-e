/**
 * @module
 */
import Random from 'random-js';
import {fromGenMaker} from '../arbitrary';
import {suchThat} from '../combinators';

/**
 * The default bound of numbers.
 *
 * @constant {number}
 */
const SIZE = 9007199254740992;

/**
 * Integer Arbitrary
 *
 * @type {Arbitrary}
 */
export let int = fromGenMaker(Random.integer, [-SIZE, SIZE]);

/**
 * Positive Integer Arbitary
 *
 * @type {Arbitary}
 */
export let pint = suchThat(int, n => n >= 0).choose(1, SIZE);

/**
 * Negative Arbitrary
 *
 * @type {Arbitrary}
 */
export let nint = suchThat(int, n => n < 0).choose(-SIZE, -1);

/**
 * Nature Number Arbitary
 *
 * @type {Arbitrary}
 */
export let nat = suchThat(int, n => n > 0);

/**
 * Nature Number Arbitary
 *
 * @type {Arbitrary}
 */
export let number = fromGenMaker(Random.real , [-SIZE, SIZE]);

/**
 * Positive Number Arbitary
 *
 * @type {Arbitrary}
 */
export let pnumber = suchThat(number, n => n >= 0).choose(1, SIZE);

/**
 * Negative Number Arbitary
 *
 * @type {Arbitrary}
 */
export let nnumber = suchThat(number, n => n < 0).choose(-SIZE, -1);
